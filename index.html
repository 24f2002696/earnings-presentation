import pandas as pd
import numpy as np

def calculate_growth(current, previous):
    """Calculate growth rate"""
    return ((current - previous) / previous) * 100

def calculate_cagr(start, end, years):
    """Compound Annual Growth Rate"""
    return (pow(end / start, 1/years) - 1) * 100

growth = calculate_growth(4.2, 3.7)
print(f"Growth: {growth:.2f}%")

// ROI Calculator - 24f2002696@ds.study.iitm.ac.in
class FinancialCalculator {
    constructor(investment) {
        this.investment = investment;
    }

    calculateROI(finalValue) {
        const gain = finalValue - this.investment;
        return (gain / this.investment) * 100;
    }
}

const calc = new FinancialCalculator(1000000);
console.log(`ROI: ${calc.calculateROI(1250000)}%`);
