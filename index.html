import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

def calculate_quarterly_growth(current, previous):
    """Calculate growth rate between quarters"""
    if previous == 0:
        return 0
    return ((current - previous) / previous) * 100

def calculate_cagr(start_value, end_value, years):
    """Compound Annual Growth Rate"""
    return (pow(end_value / start_value, 1/years) - 1) * 100

# Example usage
q3_revenue = 4.2  # billion
q2_revenue = 3.9  # billion

print(f"Q3 Growth: {growth:.2f}%")

// ROI Calculator - 24f2002696@ds.study.iitm.ac.in
class FinancialCalculator {
    constructor(initialInvestment) {
        this.investment = initialInvestment;
    }

    calculateROI(finalValue) {
        const gain = finalValue - this.investment;
        return (gain / this.investment) * 100;
    }

    calculateNPV(cashFlows, discountRate) {
        return cashFlows.reduce((npv, cf, t) => {
            return npv + cf / Math.pow(1 + discountRate, t);
        }, 0);
    }
}

const calc = new FinancialCalculator(1000000);
console.log(`ROI: ${calc.calculateROI(1250000)}%`);
